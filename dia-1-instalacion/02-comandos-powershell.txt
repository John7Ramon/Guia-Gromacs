# EJECUTAR: PowerShell como ADMINISTRADOR

# CONTEXTO: Usar cuando "Subsistema de Windows para Linux" 
# no aparece disponible en las características de Windows

# Comando 1: Habilitar el Subsistema de Windows para Linux
dism.exe /online /enable-feature /featurename:Microsoft-Windows-Subsystem-Linux /all /norestart

# Comando 2: Habilitar la Plataforma de Máquina Virtual
dism.exe /online /enable-feature /featurename:VirtualMachinePlatform /all /norestart
